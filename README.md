# CLAP-1

#include <stdio.h>
int main()
{
    int amount_given,bill_amt;
    int q,r;
    scanf("%d %d",&amount_given,&bill_amt);
    q=amount_given/bill_amt;
    
    r=amount_given%bill_amt;
   
    printf("%d is the Quotient\n",q);
    printf("%d is the Remainder",r);

    
	return 0;
}
